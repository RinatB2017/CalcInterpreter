program:
	expr_list* EXIT

expr_list:
	expr END
	void_func END
	if_

block:
	'{' expt_list '}'
	
if_:
	IF '(' expr ')' block
	IF '(' expr ')' block ELSE block
	
expr:
	term 'expr
		'expr:
			'+' term
			'-' term
			term

term:
	power 'term
		'term:
			'*' power
			'/' power
			power

power:
	factorial '^' power
	factorial
	
factorial:
	factorial '!'
	prim

prim:
	NUMBER
	TRUE
	FALSE
	USER_DEFINED_NAME
	USER_DEFINED_NAME '=' expr
	-prim
	'(' expr ')'
	func

func:
	of_radian '(' expr ')'
	returns_radian '(' expr ')'
	LOG '(' expr ')'

of_radian:
	SIN
	COS

returns_radian:
	ARCSIN
	ARCCOS

void_func:
	print
	add
	del
	reset
	set
	unset
	help
	state
	
print:
	PRINT
	PRINT expr

add:
	USER_DEFINED_NAME
	USER_DEFINED_NAME = expr
	
del:
	DEL *
	DEL USER_DEFINED_NAME
	
reset:
	RESET *
	RESET setname
	RESET TABLE
	
set:
	SET setname '=' currTok
	
setname:
	ARGS_AUTO_END, AUTO_END, PRINT_TOKENS
	PRECISION, ERRORS, STRICTED, AUTO_PRINT, GREEDY_FUNC
	
help:
	HELP
	
state:
	STATE
