program:
	expr_list* EXIT

expr_list:
	expr END
	void_func END
	if_

block:
	'{' expt_list '}'
	
if_:
	"if" '('expr')' block
	"if" '('expr')' block "else" block
expr:
	expr '+' term
	expr '-' term
	term

term:
	term '/' power
	term '*' power
	power

power:
	factorial '^' factorial
	factorial
	
factorial:
	prim '!'
	prim

prim:
	NUMBER
	USER_DEFINED_NAME
	USER_DEFINED_NAME = expr
	-prim
	(expr)
	func

func:
	sin '(' expr ')'
	cos '(' expr ')'

void_func:
	print
	add
	del
	reset
	set
	unset
	help
	state
	
print:
	PRINT
	PRINT expr

add:
	USER_DEFINED_NAME
	USER_DEFINED_NAME = expr
	
del:
	DEL *
	DEL USER_DEFINED_NAME
	
reset:

set:

help:

state:
